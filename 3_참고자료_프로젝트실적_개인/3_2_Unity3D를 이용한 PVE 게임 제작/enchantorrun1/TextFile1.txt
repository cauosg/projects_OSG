using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class SpawnHands : MonoBehaviour {
    public GameObject Player;
    public GameObject Rocks;
    public float Spawninterval = 2.0f;
    public static int Spawnnums = 10;
    private float Spawncooltime = 0.0f;
    private bool isspawning = false;
    private int spawncount = 0;
    private Vector3[] Spawnposes = new Vector3[Spawnnums];
    private bool[] Spawnbools = new bool[Spawnnums];
    private float[] eachcounts = new float[Spawnnums];
    // Use this for initialization
    void Start () {
        for (int i = 0; i < Spawnnums; i++)
        {
            eachcounts[i] = 0.0f;
            Spawnbools[i] = false;
        }

    }
	
	// Update is called once per frame
	void Update () {
        //if(Spawncooltime<0.1f)
        if (spawncount < 1)
            isspawning = true;
        if(isspawning && Spawncooltime < Spawninterval/2 && spawncount<10)
        {
            Spawnposes[spawncount] = Player.transform.position;
            spawncount++;
            Spawncooltime = Spawninterval;
        }
        if (isspawning)
        {
            for (int i = 0; i < spawncount; i++)
                spawnrocks(Spawnposes[i], i);
        }
        //if (spawncount > 0)
        //{
        //    for(int i = 0)
        //}

        //Debug.Log(spawncount);
        if (Spawncooltime > 0)
            Spawncooltime -= Time.deltaTime;
    }
    public void spawnrocks(Vector3 pos, int spawnnum)
    {
        if (!Spawnbools[spawnnum] && eachcounts[spawnnum] < 0.1f)
        {
            if (spawnnum < 1)
            {
                if (!Spawnbools[spawnnum])
                {

                    Spawnbools[spawnnum] = true;
                    eachcounts[spawnnum] = Spawninterval;
                    Instantiate(Rocks, pos, Quaternion.identity);

                }
            }
            else if (Spawnbools[spawnnum - 1] && !Spawnbools[spawnnum])
            {
                //Debug.Log("spawn");
                Spawnbools[spawnnum] = true;
                eachcounts[spawnnum] = Spawninterval;
                Instantiate(Rocks, pos, Quaternion.identity);
            }
        }
        if (Spawnbools[spawnnum])
        {
            eachcounts[spawnnum] -= Time.deltaTime;
            
        }
        if (eachcounts[spawnnum] < 0.1f)
        {
            Spawnbools[spawnnum] = false;
            
        }
    }
}
